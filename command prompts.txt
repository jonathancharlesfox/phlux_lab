$body = @{
  suction_pressure    = 2.5      # bar
  discharge_pressure = 6.0      # bar
  delta_pressure     = 3.5      # bar (can be discharge - suction or whatever your model expects)
  speed              = 1800     # rpm
  temperature        = 25.0     # degC
  fluid_density      = 1000.0   # kg/m3
  fluid_viscosity    = 1.0      # cP
  pump_power         = 10.0     # kW
  valve_opening      = 0.6      # frac
  sink_pressure      = 2.5      # bar
} | ConvertTo-Json

Invoke-RestMethod `
  -Uri "http://127.0.0.1:8000/predict" `
  -Method POST `
  -ContentType "application/json" `
  -Body $body
